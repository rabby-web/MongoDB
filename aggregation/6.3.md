## `$group, $sum, $push aggregate stage`

```js
// group
db.test.aggregate([
  // stage:-1
  {
    $group: { _id: "$address.country" },
  },
]);

// count or sum
db.test.aggregate([
  // stage:-1
  {
    $group: { _id: "$address.country", count: { $sum: 1 } },
  },
]);

// push
db.test.aggregate([
  // stage:-1
  {
    $group: {
      _id: "$address.country",
      amakaDakaoName: { $push: "$name" },
    },
  },
]);
```
