## `$group, with $unwind aggregation stage`

```js
// unwind with group and sum
db.test.aggregate([
  // stage:-1
  { $unwind: "$friends" },
  // stage:-2
  {
    $group: { _id: "$friends", count: { $sum: 1 } },
  },
]);

// unwind with group and push
db.test.aggregate([
  // stage:-1
  { $unwind: "$interests" },
  // stage:-2
  {
    $group: { _id: "$age", interestPerAge: { $push: "$interests" } },
  },
]);
```
