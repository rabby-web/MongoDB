## `$addFields, $out, $marge aggregate stage`

```js
db.test.aggregate([
  // stage:-1
  {
    $match: {
      gender: "Male",
    },
  },
  // stage:-2
  {
    // for add fields
    $addFields: {
      course: "level-2",
      eduTec: "PH",
    },
  },
  // stage:-3
  {
    $project: {
      course: 1,
      eduTec: 1,
    },
  },
  // stage:-4
  {
    // create new collection
    $out: "course-students",
  },
]);
```
