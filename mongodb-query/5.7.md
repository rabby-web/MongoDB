## `$all, $elemMatch`

```javascript
// position match show array
db.test
  .find({ interests: ["Cooking", "Writing", "Reading"] })
  .project({ interests: 1 });

// only match show array
db.test
  .find({
    interests: { $all: ["Cooking", "Writing", "Reading"] },
  })
  .project({ interests: 1 });

// position match show object
db.test
  .find({
    skills: {
      name: "JAVASCRIPT",
      level: "Intermidiate",
    },
  })
  .project({ skills: 1 });

// position match show object {isLearning: false}
db.test
  .find({
    skills: {
      name: "JAVASCRIPT",
      level: "Intermidiate",
      isLearning: false,
    },
  })
  .project({ skills: 1 });

// position match show object 
db.test
  .find({
    skills: {
      $elemMatch: {
        name: "JAVASCRIPT",
        level: "Intermidiate",
      },
    },
  })
  .project({ skills: 1 });
```
